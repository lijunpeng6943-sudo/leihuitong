// 简单的Node.js服务器，用于方言学习平台的数据读写 const http = require('http'); const fs = require('fs'); const path = require('path'); const url = require('url'); // 服务器配置 const PORT = 3000; const HOST = 'localhost'; // MIME类型映射 const mimeTypes = { '.html': 'text/html', '.js': 'application/javascript', '.css': 'text/css', '.json': 'application/json', '.mp3': 'audio/mpeg', '.wav': 'audio/wav', '.png': 'image/png', '.jpg': 'image/jpeg', '.jpeg': 'image/jpeg', '.gif': 'image/gif', '.svg': 'image/svg+xml' }; // 创建HTTP服务器 const server = http.createServer((req, res) => { const parsedUrl = url.parse(req.url, true); const pathname = parsedUrl.pathname; console.log(`请求: ${req.method} ${pathname}`); // 处理API请求 if (pathname === '/api/data' && req.method === 'GET') { handleGetData(req, res); return; } if (pathname === '/api/data' && req.method === 'POST') { handleSaveData(req, res); return; } // 处理静态文件请求 let filePath = '.' + pathname; if (filePath === './') { filePath = './index.html'; } const extname = String(path.extname(filePath)).toLowerCase(); const contentType = mimeTypes[extname] || 'application/octet-stream'; fs.readFile(filePath, (error, content) => { if (error) { if (error.code == 'ENOENT') { res.writeHead(404, { 'Content-Type': 'text/html' }); res.end('

404 Not Found

', 'utf-8'); } else { res.writeHead(500); res.end('Sorry, check with the site admin for error: ' + error.code + ' .\n'); } } else { res.writeHead(200, { 'Content-Type': contentType }); res.end(content, 'utf-8'); } }); }); // 处理获取数据请求 function handleGetData(req, res) { const dataPath = './data.json'; fs.readFile(dataPath, 'utf8', (err, data) => { if (err) { console.error('读取数据文件失败:', err); res.writeHead(500, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ error: '读取数据失败' })); return; } res.writeHead(200, { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' }); res.end(data); }); } // 处理保存数据请求 function handleSaveData(req, res) { let body = ''; req.on('data', (chunk) => { body += chunk; }); req.on('end', () => { try { const data = JSON.parse(body); const dataPath = './data.json'; fs.writeFile(dataPath, JSON.stringify(data, null, 2), 'utf8', (err) => { if (err) { console.error('保存数据文件失败:', err); res.writeHead(500, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ error: '保存数据失败' })); return; } console.log('数据保存成功!'); res.writeHead(200, { 'Content-Type': 'application/json', 'Access-Control-Allow-Origin': '*' }); res.end(JSON.stringify({ success: true, message: '数据保存成功' })); }); } catch (error) { console.error('解析数据失败:', error); res.writeHead(400, { 'Content-Type': 'application/json' }); res.end(JSON.stringify({ error: '数据格式错误' })); } }); } // 启动服务器 server.listen(PORT, HOST, () => { console.log(`\n========================================`); console.log(`方言学习平台服务器已启动!`); console.log(`访问地址: http://${HOST}:${PORT}`); console.log(`API接口:`); console.log(` - GET http://${HOST}:${PORT}/api/data (获取数据)`); console.log(` - POST http://${HOST}:${PORT}/api/data (保存数据)`); console.log(`========================================\n`); }); // 服务器错误处理 server.on('error', (error) => { if (error.code === 'EADDRINUSE') { console.error(`端口 ${PORT} 已被占用，请关闭其他占用该端口的程序后重试。`); } else { console.error('服务器启动失败:', error); } process.exit(1); }); // 优雅关闭 process.on('SIGINT', () => { console.log('\n正在关闭服务器...'); server.close(() => { console.log('服务器已关闭'); process.exit(0); }); });
